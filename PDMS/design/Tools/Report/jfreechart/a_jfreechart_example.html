jfreechart应用_例题_生成饼状图（2）_可以直接运行

Posted on 2009-01-09 11:57 liudecai_zan@126.com 阅读(14) 评论(0)  编辑  收藏

package pkg;

import org.jfree.chart.JFreeChart;
import org.jfree.chart.ChartFactory;
import org.jfree.chart.ChartFrame;
import org.jfree.data.general.DefaultPieDataset;

/**
 * 生成饼状图 只展示，不生成文件
 * 导入jar包 jaxen-1.1.1.jar，jcommon-1.0.12.jar，jfreechart-1.0.9.jar
 * @author liudecai_zan@126.com
 *
 */
public class Test {

 public static void main(String[] args) {
  DefaultPieDataset dpd = new DefaultPieDataset();
  dpd.setValue("超出实检上限总捆数", 5);
  dpd.setValue("实检范围内总捆数", 35);
  dpd.setValue("超出实检下限总捆数", 15);

  // Create JFreeChart object
  // 参数可以查看源码
  JFreeChart pieChart = ChartFactory.createPieChart(
    "棒材异常磅单超重捆数信息", dpd, true, true, false);
  ChartFrame pieFrame = new ChartFrame("棒材异常磅单超重捆数信息图", pieChart);
  pieFrame.pack();
  pieFrame.setVisible(true);

 }
}
